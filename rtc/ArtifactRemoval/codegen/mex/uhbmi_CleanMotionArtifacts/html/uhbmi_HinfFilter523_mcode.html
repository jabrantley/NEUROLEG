1function [sh,zh,Pt,WH]=uhbmi_HinfFilter(Yf, Rf, gamma, Pt, wh, q)
 2
 3%#codegen
 4
 5%warning off all
 6%Atilla Kilicarslan - 2014,

 8
 9
10
11numTP  = size(Rf,1);
12numDat = size(Yf,2);
13numRef = size(Rf,2);
14
15zh     = zeros(numTP,numDat);
16sh     = zeros(numTP,numDat);
17g      = zeros(numRef,1);
18
19for n=1:numTP         
20    r  = Rf(n,:)';
21    Ptemp  = inv(Pt) - (gamma^(-2))*(r*r');    
22    g(:,1) = (Ptemp\r)/(1+(r'/Ptemp)*r);
23    for m=1:numDat          
24        y           = Yf(n,m);
25        zh(n,m)     = r'*wh(:,m);   
26        sh(n,m)     = y-zh(n,m);   
27        wh(:,m)     = wh(:,m) + g(:,1)*sh(n,m);        
28    end 
29    Pt          = inv (  (inv(Pt))+ ((1-gamma^(-2))*(r*r')) ) + q*eye(size(Rf,2));
30    WH          = wh;
31end

33end
